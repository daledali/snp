Routine blood lead screening:
[select name="Q1" value="no|YES"] <-- moving from a foreign country within last 12 months
[select name="Q2" value="no|YES"] <-- moving from a major metropolitan area within last 12 months
[select name="Q3" value="no|YES"] <-- receiving MA (Medical Assistance)
[select name="Q4" value="no|YES"] <-- receiving WIC (Supplemental Food Program for Women, Infants, and Children)
[select name="Q5" value="no|YES"] <-- receiving Medicaid
[select name="Q6" value="no|YES"] <-- living in a home built before 1950
[select name="Q7" value="no|YES"] <-- visiting a home built before 1950 in past 6 months
[select name="Q8" value="no|YES"] <-- attending day care in a home built before 1950 in past 6 months
[select name="Q9" value="no|YES"] <-- sibling or housemate diagnosed with blood lead poisoning
[select name="Q10" value="no|YES"] <-- playmate diagnosed with blood lead poisoning
[select name="Q11" value="no|YES"] <-- living in or regularly visiting a building built before 1978 with recent repair, remodeling, or damage
